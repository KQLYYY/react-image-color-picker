## Install

```bash
npm install --save color-picker
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'color-picker'
import 'color-picker/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
